I have no name!@redis-client:/$ REDIS_PASSWORD="redis"
SENTINEL_HOST="redis"
SENTINEL_PORT="26379"
SLEEP_INTERVAL=0.1
CHECK_MASTER_INTERVAL=1  # Check master every 1 iterations

get_master() {
  MASTER_INFO=$(redis-cli -h $SENTINEL_HOST -p $SENTINEL_PORT -a $REDIS_PASSWORD SENTINEL get-master-addr-by-name mymaster)
  MASTER_HOST=$(echo $MASTER_INFO | awk '{print $1}')
  MASTER_PORT=$(echo $MASTER_INFO | awk '{print $2}')
}

get_master
COUNTER=0

while true; do
  if [ $((COUNTER % CHECK_MASTER_INTERVAL)) -eq 0 ]; then
    get_master
  fi
  TIMESTAMP=$(date +"%Y-%m-%d %H:%M:%S")
  KEY="key$(date +%s%N)"
  VALUE="value$(date +%s%N)"
  redis-cli -h $MASTER_HOST -p $MASTER_PORT -a $REDIS_PASSWORD SET $KEY $VALUE
  if [ $? -eq 0 ]; then
    echo "$TIMESTAMP: Set $KEY to $VALUE"
  else
    echo "$TIMESTAMP: Failed to set $KEY" >&2
    get_master  # Re-check master on failure
  fi
  sleep $SLEEP_INTERVAL
  COUNTER=$((COUNTER + 1))
done
